int main() {
    int i, j, n, count;

    printf("Enter number of array elements: ");
    scanf("%d", &n);

    int arr[n];
    printf("Enter array elements: ");
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    printf("Elements that appear only once: ");
    for (i = 0; i < n; i++) {
        count = 0;
        for (j = 0; j < n; j++) {
            if (arr[i] == arr[j]) {
                count++;
            }
        }
        if (count == 1) {
            printf("%d ", arr[i]);
        }
    }

    return 0;
}

2)sorted array missing element
#include <stdio.h>

int main() {
    int i, n;

    printf("Enter number of array elements: ");
    scanf("%d", &n);

    int arr[n];
    printf("Enter %d sorted distinct array elements:\n", n);
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    for (i = 0; i < n; i++) {
        if (arr[i] != i) {
            printf("Smallest missing element is: %d\n", i);
            return 0;
        }
    }

    printf("Smallest missing element is: %d\n", n);
    return 0;
}
