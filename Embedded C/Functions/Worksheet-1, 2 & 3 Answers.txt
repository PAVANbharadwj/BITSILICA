Expected output: 34 = 3 + 31

#include <stdio.h>

int isPrime(int num) {
    if (num <= 1) return 0;
    for (int i = 2; i * i <= num; i++) {
        if (num % i == 0)
            return 0;
    }
    return 1;
}

void primes(int n) {
    int found = 0;
    for (int i = 2; i <= n / 2; i++) {
        if (isPrime(i) && isPrime(n - i)) {
            printf("The 2 prime numbers of %d is %d + %d\n", n, i, n - i);
            found = 1;
            break;
        }
    }
    if (!found) {
        printf("%d There is NO sum of two prime numbers.\n", n);
    }
}

int main() {
    int n;
    printf("Enter a number: ");
    scanf("%d", &n);
    primes(n);
    return 0;
}

9. Convert a binary number to decimal and vice-versa
Question: Write functions to convert binary to decimal and decimal to binary.
Sample data: Binary input: 1010, Decimal input: 10
Expected output: Binary 1010 = Decimal 10, Decimal 10 = Binary 1010


#include <stdio.h>
#include <math.h>


int binaryToDecimal(long long bin) {
    int dec = 0, base = 1;
    while (bin > 0) {
        int rem = bin % 10;
        dec += rem * base;
        base *= 2;
        bin /= 10;
    }
    return dec;
}

long long decimalToBinary(int dec) {
    long long bin = 0;
    int rem, i = 1;
    while (dec != 0) {
        rem = dec % 2;
        bin += rem * i;
        dec /= 2;
        i *= 10;
    }
    return bin;
}

int main() {
    long long bin;
    int dec;

    printf("Enter a binary number: ");
    scanf("%lld", &bin);
    printf("Enter a decimal number: ");
    scanf("%d", &dec);
    int decRes = binaryToDecimal(bin);
    printf("Binary %lld = Decimal %d\n", bin, decRes);
    long long binRes = decimalToBinary(dec);
    printf("Decimal %d = Binary %lld\n", dec, binRes);

    return 0;
}

Question: Write a program in C to check whether two given strings are an anagram.
Sample data: String1: spare
String2: pears
Expected output: spare and pears are Anagram.


#include <stdio.h>
#include<string.h>

void sort(char str[]){
    int n=strlen(str);
    for(int i=0;i<n-1;i++){
        for(int j=i+1;j<n;j++){
            if (str[i] > str[j]) {
                char temp = str[i];
                str[i] = str[j];
                str[j] = temp;
            }
        }
    }
}

int anagram(char str1[],char str2[]){
    if(strlen(str1)!=strlen(str2)){
        return 0;
    }
    sort(str1);
    sort(str2);
    
    return strcmp(str1,str2)==0;
}

int main() {
    char str1[100],str2[100];
    printf("Enter str1: ");
    scanf("%s",str1);
    printf("Enter str2: ");
    scanf(" %s",str2);
    if(anagram(str1,str2)){
        printf("The 2 strings are anagrams.");
    }
    else{
        printf("The 2 strings are not anagram");
    }
    return 0;
}

Write a variadic function that calculates the average of any number of integers.

#include <stdio.h>
#include <stdarg.h>

double average(int count, ...) {
    va_list args;
    va_start(args, count);

    int sum = 0;
    for (int i = 0; i < count; i++) {
        sum += va_arg(args, int);  
    }

    va_end(args);

    return count == 0 ? 0 : (double)sum / count; 
}

int main() {
    printf("Average: %.2f\n", average(4, 10, 20, 30, 40));
    printf("Average: %.2f\n", average(3, 5, 15, 25));
    return 0;
}
