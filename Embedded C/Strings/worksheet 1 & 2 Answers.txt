palindromic substring
'#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    int n, i, j;
    int dp[100];       
    int isPal[100][100]; 
    printf("Enter a string: ");
    scanf("%s", str);
    n = strlen(str);

    for (i = 0; i < n; i++) {
        for (j = i; j < n; j++) {
            int l = i, r = j;
            int flag = 1;
            while (l < r) {
                if (str[l] != str[r]) {
                    flag = 0;
                    break;
                }
                l++; r--;
            }
            isPal[i][j] = flag;
        }
    }

    for (i = 0; i < n; i++) {
        if (isPal[0][i]) {
            dp[i] = 0;
        } else {
            dp[i] = i;
            for (j = 1; j <= i; j++) {
                if (isPal[j][i] && dp[j - 1] + 1 < dp[i]) {
                    dp[i] = dp[j - 1] + 1;
                }
            }
        }
    }

    printf("Minimum palindromic substrings: %d\n", dp[n - 1] + 1);
    return 0;
}
